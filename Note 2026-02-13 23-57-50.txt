  const name = document.getElementById("nameInput").value.trim();

  }

  const encoded = encodeURIComponent(name);

    <a href="${link}" target="_blank">${link}</a>

    <button onclick="shareWhatsApp('${link}')">Share on WhatsApp</button>
  `;
}
